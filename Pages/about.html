    import java.util.*;
    import java.util.concurrent.*;
    
    /**
     * Class representing the user's personal bio and its related methods.
     */
    public class AboutMe {
        private String bio;
        private ExecutorService executorService;
    
        public AboutMe() {
            this.bio = "🚀 Passionate lifelong learner, always exploring emerging tech and innovative solutions.\n"
                    + "👨‍💻 Skilled in Agile Scrum, actively collaborating in daily scrums, sprint reviews, and release planning to drive team success.\n"
                    + "🧠 Strong problem-solver with a knack for debugging and delivering robust tech solutions.\n"
                    + "✅ Proactive in testing and quality assurance, with experience in Test-driven Development (TDD) and using tools like JUnit for unit tests.\n"
                    + "🌐 Extensive experience in developing RESTful APIs with Spring Boot, and proficient in Git for version control.\n"
                    + "💻 Deep understanding of Java, OOP principles, Java 8 features, and MVC architecture, alongside hands-on expertise in Spring Framework and database integration.\n"
                    + "🌱 Familiar with full-stack web development, including frontend technologies like HTML, CSS, and JavaScript, with a solid grasp of basic MVC and database design.";
            this.executorService = Executors.newFixedThreadPool(3);
        }
    
        public String getBio() {
            return bio;
        }
    
        public void printBio() {
            System.out.println("Bio: " + this.bio);
        }
    
        // Simulating concurrent tasks
        public void simulateConcurrentTasks() {
            Runnable task1 = () -> {
                try {
                    Thread.sleep(2000);  // Simulate a delay
                    System.out.println("Fetching Bio...");
                    printBio();
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            };
    
            executorService.submit(task1);
        }
    
        public void shutDown() {
            executorService.shutdown();
        }
    }

    /**
     * Class to store and display experience and roles.
     */
    class Experience {
        private List roles;
        private List companies;
        private List timeLine;
    
        public Experience() {
            this.roles = Arrays.asList("Java Developer", "Blockchain Developer");
            this.companies = Arrays.asList("GoDigit General Insurance", "GTPL Hathway");
            this.timeLine = Arrays.asList("Jun 2023 - Present", "Feb 2023 - May 2023");
        }
    
        public void printExperience() {
            System.out.println("Experience:");
            for (int i = 0; i < roles.size(); i++) {
                System.out.println("Role: " + this.roles.get(i));
                System.out.println("Company: " + this.companies.get(i));
                System.out.println("Time: " + this.timeLine.get(i));
                System.out.println();
            }
        }
    }
    
    /**
     * Main class to execute the application and simulate the output.
     */
    public class Main {
        public static void main(String[] args) {
            AboutMe aboutMe = new AboutMe();
            SkillsAndTechnologies skills = new SkillsAndTechnologies();
            Experience experience = new Experience();
    
            // Simulate tasks with concurrency
            aboutMe.simulateConcurrentTasks();
    
            // Simulate delay to print skills and experience after bio
            try {
                Thread.sleep(10000);
                skills.printSkills();
                experience.printExperience();
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            } finally {
                aboutMe.shutDown();
            }
        }
    }
